293        /**
294         * Creates a model handler for a model name.
295         *
296         * @param modelName
297         * @return
298         */
299        private ModelHandler createModelHandler(String modelName) {
300            if(modelHandlers.containsKey(modelName)) {
301                throw new IllegalArgumentException("a model with name " + modelName + " already exists.");   
302            }
303    
304            CodeModelBase cmb = coderFactory.createCodeModel();
305            CodeHandler   ch  = coderFactory.createHandlerOnModel(cmb);
306            JavaQueryModel qm = coderFactory.createQueryModel(cmb);
307            ModelHandler mh   = new ModelHandler(cmb, ch, qm);
308            modelHandlers.put(modelName, mh);
309            return mh;
310        }

312        /**
313         * Removes a model handler for the specified model name.
314         *
315         * @param modelName
316         */
317        private void removeModelHandler(String modelName) {
318            if( ! modelHandlers.containsKey(modelName)) {
319                throw new IllegalArgumentException("model with name " + modelName + " doesn't exist."); 
320            }
321            if(modelHandlers.size() == 1) {
322                throw new IllegalArgumentException("at least a model must be present in the models set");
323            }
324    
325            ModelHandler mh = modelHandlers.get(modelName);
326            mh.codeModel.clearAll();
327            modelHandlers.remove(modelName);
328        }

357        /**
358         * Performs a SPARQL query on the specified model.
359         * 
360         * @param modelName
361         * @param qry
362         * @param ps
363         */
364        private void performQueryOnModel(String modelName, String qry, PrintStream ps) {
365            if( ! modelHandlers.containsKey(modelName)) {
366                throw new IllegalArgumentException("model with name " + modelName + " doesn't exist.");
367            }
368    
369            ModelHandler mh = modelHandlers.get(modelName);
370            CodeModel cm = mh.codeModel;
371            if( ! (cm instanceof SPARQLQuerableCodeModel) ) {
372                throw new IllegalArgumentException("code model " + modelName + " is not a QuerableCodeModel instance.");
373            }
374            SPARQLQuerableCodeModel qcm = (SPARQLQuerableCodeModel) cm;
375            QueryResult qr = null;
376            try {
377                qr = qcm.performQuery(qry);
378                qr.toTabularView(ps);
379            } catch (SPARQLException e) {
380                throw new IllegalArgumentException("Cannot perform SPARQL query.");
381            } finally {
382                if(qr != null) { qr.close(); }
383            }
384        }

397         * Inspect the specified model as a bean.
398         *
399         * @param modelName
400         * @param qry
401         * @param ps
402         */
403        private void inspectModel(String modelName, String qry, PrintStream ps) {
404            if( ! modelHandlers.containsKey(modelName) ) {
405                throw new IllegalArgumentException("model with name '" + modelName + "' doesn't exist.");
406            }
407    
408            ModelHandler mh = modelHandlers.get(modelName);
409            JavaQueryModel qm = mh.queryModel;
410    
411            if( inspector == null) {
412                inspector = new Inspector();
413                inspector.addToContext(INSPECTION_MODEL_NAME, qm);
414            }
415    
416            try {
417                Object o = inspector.inspect(qry);
418                ps.println(o);
419            } catch (Exception e) {
420                throw new IllegalArgumentException("Cannot perform inspection query.", e);
421            }
422        }

593        /**
594         * Loads a list of libraries and reports the result operations on the given PrintStream.
595         * 
596         * @param args
597         * @param ps
598         * @throws IOException
599         */
600        private void loadLibraries(String[] args, PrintStream ps) throws IOException {
601            String libraryName;
602    
603            // Validate arguments first.
604            List<Library> libraries = new ArrayList<Library>();
605            for(int i = 0; i < args.length; i += 2) {
606                libraryName = args[i];
607                validLibraryName(libraryName);
608                String resource = args[i + 1];
609                File resourceFile;
610                LibraryType type;
611                if( (resourceFile = validJar(resource)) != null ) {
612                    type = LibraryType.JAR_FILE;
613                } else if ( (resourceFile = validSource(resource) ) != null ) {
614                    type = LibraryType.SOURCE_DIR;
615                } else if ( (resourceFile = validJavadoc(resource)) != null ) {
616                    type = LibraryType.JAVADOC_DIR;
617                } else if ( (resourceFile = validClass(resource)  ) != null ) {
618                    type = LibraryType.CLASS_DIR;
619                } else {
620                    throw new IllegalArgumentException("Cannot find resource: '" + args[i + 1] + "'");
621                }
622                libraries.add(new Library(libraryName, resourceFile, type));
623            }
624    
625            // Process arguments.
626            ObjectsTable ot = new ObjectsTable();
627            JStatistics statistics = new JStatistics();
628            CodeHandler statisticsCodeHandler = statistics.createStatisticsCodeHandler( getCodeHandler() );
629    
630            JavaBytecodeJarParser javaBytecodeJarParser = null;
631            DirectoryParser sourceDirectoryParser       = null;
632            DirectoryParser javadocDirectoryParser      = null;
633            DirectoryParser classDirectoryParser        = null;
634            for(int i = 0; i < libraries.size(); i++) {
635    
636                try {
637                    //statisticsCodeHandler.startParsing(libraries.get(i).name, libraries.get(i).location.getAbsolutePath());
638    
639                    javaBytecodeJarParser = new JavaBytecodeJarParser();
640                    javaBytecodeJarParser.initialize( statisticsCodeHandler, ot );
641    
642                    JavaSourceFileParser javaSourceFileParser = new JavaSourceFileParser();
643                    sourceDirectoryParser = new DirectoryParser(javaSourceFileParser, new CoderUtils.JavaSourceFilenameFilter() );
644                    sourceDirectoryParser.initialize( statisticsCodeHandler, ot );
645    
646                    JavadocFileParser javadocFileParser = new JavadocFileParser();
647                    javadocDirectoryParser = new DirectoryParser(javadocFileParser, new CoderUtils.JavaSourceFilenameFilter() );
648                    javadocDirectoryParser.initialize( statisticsCodeHandler, ot );
649    
650                    JavaBytecodeFileParser javaBytecodeFileParser = new JavaBytecodeFileParser();
651                    classDirectoryParser = new DirectoryParser(javaBytecodeFileParser, new CoderUtils.JavaClassFilenameFilter() );
652                    classDirectoryParser.initialize( statisticsCodeHandler, ot );
653    
654                    if( libraries.get(i).type == LibraryType.JAR_FILE ) {
655    
656                        try {
657                            System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
658                            javaBytecodeJarParser.parseFile(libraries.get(i).location);
659                            System.out.println(" done");
660                        } catch (Exception e) {
661                            throw new IllegalArgumentException("Error while loading JAR: '" + libraries.get(i).location + "'", e);
662                        }
663    
664                    } else if( libraries.get(i).type == LibraryType.SOURCE_DIR ) {
665    
666                        try {
667                            System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
668                            sourceDirectoryParser.parseDirectory( libraries.get(i).name, libraries.get(i).location );
669                            System.out.println(" done");
670                        } catch(Exception e) {
671                            throw new IllegalArgumentException("Error while reading SRC dir: '" + libraries.get(i).location + "'", e);
672                        }
673    
674                    } else if(libraries.get(i).type == LibraryType.JAVADOC_DIR ) {
675    
676                        try {
677                            System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
678                            javadocDirectoryParser.parseDirectory( libraries.get(i).name, libraries.get(i).location );
679                            System.out.println(" done");
680                        } catch (Exception e) {
681                            throw new IllegalArgumentException("Error while reading JAVADOC dir: '" + libraries.get(i).location + "'", e);
682                        }
683    
684                    } else if(libraries.get(i).type == LibraryType.CLASS_DIR ) {
685    
686                        try {
687                            System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
688                            classDirectoryParser.parseDirectory( libraries.get(i).name, libraries.get(i).location );
689                            System.out.println(" done");
690                        } catch (Exception e) {
691                            throw new IllegalArgumentException("Error while reading CLASS dir: '" + libraries.get(i).location + "'", e);
692                        }
693    
694                    } else {
695                        throw new IllegalStateException("Cannot load library: '" + libraries.get(i).name + "'");
696                    }
697    
698                } finally {
699                    if(javaBytecodeJarParser  != null) { javaBytecodeJarParser.dispose();  }
700                    if(sourceDirectoryParser  != null) { sourceDirectoryParser.dispose();  }
701                    if(javadocDirectoryParser != null) { javadocDirectoryParser.dispose(); }
702                    if(classDirectoryParser   != null) { classDirectoryParser.dispose();   }
703                }
704            }
705    
706            //statisticsCodeHandler.endParsing();
707    
708            ot.clear();
709    
710            ps.print(statistics.toStringReport());
711            statistics.clean();
712    
713            toBeSaved.add(selectedModel);
714        }

742         *
743         * @param param
744         * @param args
745         * @return
746         */
747        private String retrieveParam(String param, String[] args) {
748            String target = param + "=";
749            for(String arg : args) {

751                    return arg.substring(target.length());
752                }
753            }
754            return null;

765        /**
766         * Retrieves the short command description of a specified command.
767         *
768         * @param commandName
769         * @return
770         * @throws IllegalAccessException
771         * @throws InvocationTargetException
772         */
773        private String getShortCommandDescription(String commandName) throws IllegalAccessException, InvocationTargetException {
774            Method[] methods = this.getClass().getMethods();
775            for(int i = 0; i < methods.length; i++) {

779                }
780            }
781            return null;
782        }

807         * @throws IllegalAccessException
808         * @throws InvocationTargetException
809         */
810        private Command[] getAvailableCommands() throws IllegalAccessException, InvocationTargetException {
811            Method[] methods = this.getClass().getMethods();
812            List<Command> commands = new ArrayList<Command>();
813            for(int i = 0; i < methods.length; i++) {
814                Command command;

816                    String commandName =  methods[i].getName().substring(COMMAND_PREFIX.length());

1340        /**
1341         * Processes a command line input.
1342         *
1343         * @param args
1344         * @throws IllegalAccessException
1345         * @throws InvocationTargetException
1346         */
1347        protected boolean processCommand(String[] args) throws IllegalAccessException, InvocationTargetException {
1348            if(args.length == 0) {
1349                return true;
1350            }
1351    
1352            String command = args[0];
1353            String[] commandArgs = new String[args.length -1];
1354            int i;
1355            for(i = 1; i < args.length; i++) {
1356                commandArgs[i - 1] = args[i];
1357            }
1358            Method[] methods = this.getClass().getMethods();
1359            for(i = 0; i < methods.length; i++) {
1360                if( methods[i].getName().equals( COMMAND_PREFIX + command)) {
1361                    try {
1362                        methods[i].invoke(this, (Object) commandArgs);
1363                    } catch (InvocationTargetException ite) {
1364                        if (ite.getCause() instanceof IllegalArgumentException) {
1365                            IllegalArgumentException iae = (IllegalArgumentException) ite.getCause();
1366                            handleIllegalArgumentException(iae, command);
1367                            return false;
1368                        } else {
1369                            throw ite;
1370                        }
1371                    }
1372                    return true;
1373                }
1374            }
1375            throw new IllegalArgumentException("unknown command: " + command);
1376        }

1378        /**
1379         * Handles an IllegalArgumentException raised by a command method transforming it
1380         * in a error message.
1381         * 
1382         * @param iae
1383         * @param cmd
1384         */
1385        private void handleIllegalArgumentException(IllegalArgumentException iae, String cmd) {
1386            System.out.println("ERROR: '" + iae.getMessage() + "'");
1387            if(debug) { iae.printStackTrace(); }
1388            Throwable cause = iae.getCause();
1389            int causeLevel = 0;
1390            while(cause != null) {
1391                System.out.println("[" + (causeLevel++) + "] with cause: '" + iae.getCause().getMessage() + "'");
1392                if(debug) { iae.getCause().printStackTrace(); }
1393                cause = cause.getCause();
1394            }
1395    
1396            try {
1397                if(cmd == null) {
1398                    printUsage(System.out);
1399                } else {
1400                    System.out.println(getLongCommandDescription(cmd));
1401                }
1402            } catch (IllegalAccessException e) {
1403                e.printStackTrace();
1404            } catch (InvocationTargetException e) {
1405                e.printStackTrace();
1406            }
1407        }

1491            printHello();
1492            while (true) {
1493                String[] arguments = extractArgs( readInput( getPrompt() ) );
1494                if ( isExit(arguments) && confirmExit() ) {
1495                    System.out.println("Bye");
1496                    System.exit(0);
1497                }
1498                try {
1499                    processCommand(arguments);
1500                } catch (IllegalArgumentException iae) {
1501                    handleIllegalArgumentException(iae);
1502                } catch (Throwable t) {
1503                    handleGenericException(t);
1504                }
1505            }
1506        }

1508        private static List<String> commands = new ArrayList();
1509    
1510        /**
1511         * Extracts arguments from a command line input.
1512         *
1513         * @param cl
1514         * @return the list of arguments.
1515         */
1516        protected static String[] extractArgs(String cl) {
1517            commands.clear();
1518            boolean insideQuotes = false;
1519            int begin = 0;
1520            for(int c = 0; c < cl.length(); c++) {
1521                if( cl.charAt(c) == '"' ) {
1522                    if(insideQuotes) {
1523                        insideQuotes = false;
1524                        if(c - begin > 0) {
1525                            commands.add(cl.substring(begin, c));
1526                        }
1527                    } else {
1528                        insideQuotes = true;
1529                    }
1530                    begin = c + 1;
1531                }
1532                if(insideQuotes) {
1533                    if(c == cl.length() - 1) {
1534                        throw new IllegalArgumentException("not found quotes closure.");
1535                    }
1536                    continue;
1537                }
1538                if( cl.charAt(c) == ' ' ) {
1539                    if(c - begin > 0) {
1540                        commands.add( cl.substring(begin, c) );
1541                    }
1542                    begin = c + 1;
1543                } else if(c == cl.length() - 1 && c - begin > 0) {
1544                    commands.add( cl.substring(begin, c + 1) );
1545                }
1546            }
1547            return commands.toArray(new String[commands.size()]);
1548        }

1551         * Returns <code>true</code> if <i>EXIT_COMMAND</i> has been specified.
1552         * @param args
1553         * @return
1554         */
1555        private static boolean isExit(String[] args) {
1556            return args.length > 0 && args[0].equals(EXIT_COMMAND);
1557        }
