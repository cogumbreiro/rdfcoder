019    package com.asemantics.model;
020    
021    import com.asemantics.sourceparse.JavadocEntry;
022    import com.asemantics.sourceparse.ObjectsTable;
023    
024    /**
025     * A <code>CodeHandler</code> implementation
026     * printing on System.out all the received events.
027     */
028    public class CodeHandlerDebugImpl implements CodeHandler {
029    
030        public void startParsing(String libraryName, String location) {
031            System.out.println("Start parsing process of library '" + libraryName + "' at location '" + location + "'");
032        }
033    
034        public void endParsing() {
035            System.out.println("End parsing process");
036        }
037    
038        public void startCompilationUnit(String identifier) {
039            System.out.println("Start compilation unit: " + identifier);
040        }
041    
042        public void endCompilationUnit() {
043            System.out.println("End compilation unit");
044        }
045    
046        public void startPackage(String pathToPackage) {
047            System.out.println("Start package: " + pathToPackage);
048        }
049    
050        public void endPackage() {
051            System.out.println("End package");
052        }
053    
054        public void startInterface(String pathToInterface, String[] extendedInterfaces) {
055            System.out.print("Start interface " + pathToInterface + " extending < ");
056            for(int i = 0; i < extendedInterfaces.length; i++) {
057                System.out.print(" " +  extendedInterfaces[i] + " ");
058            }
059            System.out.println(">");
060        }
061    
062        public void endInterface() {
063            System.out.println("End interface");
064        }
065    
066        public void startClass(
067                JavaCodeModel.JModifier[] modifiers,
068                JavaCodeModel.JVisibility visibility,
069                String pathToClass,
070                String extendedClass,
071                String[] implementedInterfaces
072        ) {
073            System.out.print(
074                    "Start class " + pathToClass +
075                            " visibility: " + visibility +
076                            " modifiers: " + JavaCodeModel.JModifier.toByte(modifiers)
077            );
078            System.out.print(" extending class " + extendedClass);
079            System.out.print(" implemented interfaces < ");
080            for(int i = 0; i < implementedInterfaces.length; i++) {
081                System.out.print(" " + implementedInterfaces[i] + " ");
082            }
083            System.out.print(">");
084        }
085    
086        public void endClass() {
087            System.out.println("end class");
088        }
089    
090        public void startEnumeration(
091                JavaCodeModel.JModifier[] modifiers,
092                JavaCodeModel.JVisibility visibility,
093                String pathToEnumeration,
094                String[] elements
095        ) {
096            System.out.print(
097                    "start enumeration " + pathToEnumeration +
098                            " visibility " + visibility  +
099                            " modifiers "  + JavaCodeModel.JModifier.toByte(modifiers)
100            );
101            System.out.println("{");
102            for(int i = 0; i < elements.length; i++) {
103                System.out.println(elements[i]);
104            }
105            System.out.println("}");
106        }
107    
108        public void endEnumeration() {
109            System.out.println("end enumeration");
110        }
111    
112        public void attribute(
113                JavaCodeModel.JModifier[] modifiers,
114                JavaCodeModel.JVisibility visibility,
115                String pathToAttribute,
116                JavaCodeModel.JType type,
117                String value) {
118            System.out.print(
119                    "attribute " + pathToAttribute +
120                            " visibility " + visibility +
121                            " modifiers " + JavaCodeModel.JModifier.toByte(modifiers)
122            );
123            System.out.print(" of type " + type + " ");
124            System.out.println(" with value " + value);
125        }
126    
127        public void constructor(
128                JavaCodeModel.JModifier[] modifiers,
129                JavaCodeModel.JVisibility visibility,
130                int overloadIndex,
131                String[] parameterNames,
132                JavaCodeModel.JType[] parameterTypes,
133                JavaCodeModel.ExceptionType[] exceptions
134        ) {
135            System.out.print(
136                    "constructor " + overloadIndex +
137                            " visibility " + visibility +
138                            " modifiers "  + JavaCodeModel.JModifier.toByte(modifiers)
139            );
140            System.out.println("{");
141            printParameters(parameterNames, parameterTypes);
142            System.out.println("}");
143            printExceptions(exceptions);
144        }
145    
146        public void method(
147                JavaCodeModel.JModifier[] modifiers,
148                JavaCodeModel.JVisibility visibility,
149                String pathToMethod,
150                int overloadIndex,
151                String[] parameterNames,
152                JavaCodeModel.JType[] parameterTypes,
153                JavaCodeModel.JType returnType,
154                JavaCodeModel.ExceptionType[] exceptions
155        ) {
156            System.out.print(
157                    "method " + pathToMethod + " index " + overloadIndex +
158                            " visibility " + visibility +
159                            " modifiers " + JavaCodeModel.JModifier.toByte(modifiers)
160            );
161            System.out.println(" return type " + returnType);
162            System.out.println("{");
163            printParameters(parameterNames, parameterTypes);
164            System.out.println("}");
165            printExceptions(exceptions);
166        }
167    
168        public void parseError(String location, String description) {
169            System.out.println("parseError: location: " + location + " description: " + description);
170        }
171    
172        public void unresolvedTypes(String[] types) {
173            System.out.println("unresolvedTypes {");
174            for(int i = 0; i < types.length; i++) {
175                System.out.println(types[i]);
176            }
177            System.out.println("}");
178        }
179    
180        public void preloadObjectsFromModel(ObjectsTable objectsTable) {
181            throw new UnsupportedOperationException();
182        }
183    
184        public void serializeUnresolvedTypeEntries(ObjectsTable objectTable) {
185            throw new UnsupportedOperationException();
186        }
187    
188        public void deserializeUnresolvedTypeEntries(ObjectsTable objectTable) {
189            throw new UnsupportedOperationException();
190        }
191    
192        public void addErrorListener(ErrorListener errorListener) {
193            throw new UnsupportedOperationException();
194        }
195    
196        public void removeErrorListener(ErrorListener errorListener) {
197            throw new UnsupportedOperationException();
198        }
199    
200        public void parsedEntry(JavadocEntry entry) {
201            System.out.println("parsedEntry: " + entry );
202        }
203    
204        public void classJavadoc(JavadocEntry entry, String pathToClass) {
205            System.out.println("classJavadoc:" + entry + " at location: " + pathToClass );
206        }
207    
208        public void methodJavadoc(JavadocEntry entry, String pathToMethod, String[] signature) {
209            System.out.println("methodJavadoc: " + entry + " at location: " + pathToMethod);
210            System.out.print("(");
211            for(String s: signature) { System.out.print(s + ", "); }
212            System.out.println(")");
213        }
214    
215        public String generateTempUniqueIdentifier() {
216            return "debug_identifier_" + System.currentTimeMillis();
217        }
218    
219        public int replaceIdentifierWithQualifiedType(String identifier, String qualifiedType) {
220            System.out.println("replaceIdentifierWithQualifiedType " + identifier + " => " + qualifiedType);
221            return 0;
222        }
223    
224        private void printParameters(String[] names, JavaCodeModel.JType[] types) {
225            for(int i = 0; i < names.length; i++) {
226                System.out.println(names[i] + ": " + types[i]);
227            }
228        }
229    
230        private void printExceptions(JavaCodeModel.ExceptionType[] exceptions) {
231            if(exceptions == null || exceptions.length == 0) {
232                return;
233            }
234            System.out.print(" throws ");
235            System.out.print("<");
236            for(int i = 0; i < exceptions.length; i++) {
237                System.out.print(" " + exceptions[i] + " ");
238            }
239            System.out.println(">");
240        }
241    }
