151        /**
152         * Checks that in the current repository a <i>Java Runtime Environment</i>
153         * model already exists.
154         *
155         * @param pathToJRE
156         * @return <code>true</code> if the resource is already defined.
157         */
158        public boolean checkJREInit(File pathToJRE) {
159            if( pathToJRE == null ) {
160                throw new RDFCoderException("pathToJRE cannot be null.");
161            }
162            if( ! pathToJRE.exists() ) {
163                throw new RDFCoderException("specified JRE path: '" + pathToJRE.getAbsolutePath() + "' doesn't exist.");
164            }
165    
166            String jreResourceName = getJREModelResourceName( pathToJRE.getName() );
167            return repository.containsResource(jreResourceName);
168        }
169    
170        public JREReport initJRE(File pathToJRE) {
171            if( pathToJRE == null ) {
172                throw new RDFCoderException("pathToJRE cannot be null.");
173            }
174            if( ! pathToJRE.exists() ) {
175                throw new RDFCoderException("specified JRE path: '" + pathToJRE.getAbsolutePath() + "' doesn't exist.");
176            }
177    
178            // Prepares jars list.
179            File[] libJars = new File( pathToJRE, JRE_LIB_DIR).listFiles( new CoderUtils.JavaJarFilter() );
180            File[] libExts = new File( pathToJRE, JRE_EXT_DIR).listFiles( new CoderUtils.JavaJarFilter() );
181            File[] libSecs = new File( pathToJRE, JRE_SEC_DIR).listFiles( new CoderUtils.JavaJarFilter() );
182            List<File> files = new ArrayList<File>(libJars.length + libExts.length + libSecs.length);
183            files.addAll( Arrays.asList(libJars) );
184            files.addAll( Arrays.asList(libExts) );
185            files.addAll( Arrays.asList(libSecs) );
186    
187            // Creates structures.
188            JStatistics statistics    = new JStatistics();
189            ObjectsTable objectsTable = model.getObjectsTable();
190            CodeModelBase cmb         = model.getCodeModelBase();
191            JavaCodeHandler javaCodeHandler = model.getCoderFactory().createHandlerOnModel(cmb);
192    
193            CodeStorage cs = model.getCoderFactory().createCodeStorage();
194            if( ! cs.supportsFile() ) {
195                throw new IllegalStateException();
196            }
197    
198            // Initializes structures.
199            final String jreName = pathToJRE.getName();
200            JavaBytecodeJarParser parser = new JavaBytecodeJarParser();
201            JavaCodeHandler statCH = statistics.createStatisticsCodeHandler(javaCodeHandler);
202            parser.initialize(statCH, objectsTable);
203            javaCodeHandler.startParsing( jreName, pathToJRE.getAbsolutePath() );
204    
205            // Does jar parsing.
206            for(File f : files) {
207                try {
208                    parser.parseFile(f);
209                } catch (Exception e) {
210                    logger.error( String.format("Error while parsing file [%s]", f.getAbsolutePath() ), e );
211                }
212            }
213    
214            // Disposes parsing.
215            javaCodeHandler.endParsing();
216    
217            // Stores the objects table content.
218            try {
219                serializeJREObjectsTable(jreName, objectsTable);
220            } catch (Exception e) {
221                throw new RDFCoderException("Cannot serialize Object Table in repository.", e);   
222            }
223    
224            objectsTable = null;
225            parser.dispose();
226            parser = null;
227    
228            // Saves model in repository.
229            OutputStream os = null;
230            Repository.Resource resource = null;
231            try {
232                resource = repository.createResource(
233                        getJREModelResourceName(pathToJRE.getName()),
234                        Repository.ResourceType.XML
235                );
236                os  = resource.getOutputStream();
237                codeStorage.saveModel(cmb, os);
238                os.close();
239            } catch (Exception e) {
240                resource.delete();
241                throw new RDFCoderException("Cannot store model in repository.", e);
242            }
243    
244            // Disposes model.
245            cmb.clearAll();
246            cmb = null;
247    
248            return new JREReport(pathToJRE ,statistics);
249        }

251        /**
252         * Loads an existing JRE data.
253         *
254         * @param pathToJRE
255         */
256        public void loadJRE(File pathToJRE) {
257            try {
258                model.getObjectsTable().load( deserializeJREObjectsTable( pathToJRE.getName() ) );
259            } catch (Exception e) {
260                throw new RDFCoderException("Cannot deserialize Object Table from repository.", e);
261            }
262        }

272        public JStatistics loadJar(String libName, String pathToJar) throws IOException, ParserException {
273            JStatistics statistics = new JStatistics();
274            JavaCodeHandler javaCodeHandler = model.getCoderFactory().createHandlerOnModel( model.getCodeModelBase() );
275    
276            JavaBytecodeJarParser parser = new JavaBytecodeJarParser();
277            JavaCodeHandler statCH = statistics.createStatisticsCodeHandler(javaCodeHandler);
278            parser.initialize( statCH, model.getObjectsTable() );
279    
280            File lib = new File(pathToJar);
281    
282            javaCodeHandler.startParsing(libName, lib.getAbsolutePath());
283            parser.parseFile( new File(pathToJar) );
284            javaCodeHandler.endParsing();
285    
286            statistics.reset();
287            parser.dispose();
288            parser = null;
289    
290            return statistics;
291        }

293        /**
294         * Loads Java resources.
295         * 
296         * @param libName
297         * @param path
298         * @param fileParser
299         * @return
300         */
301        private JStatistics loadJava(String libName, String path, FileParser fileParser) {
302             JavaCodeHandler ch = model.getCoderFactory().createHandlerOnModel( model.getCodeModelBase() );
303    
304            DirectoryParser directoryParser = new DirectoryParser( fileParser, new CoderUtils.JavaSourceFilenameFilter() );
305            JStatistics statistics = new JStatistics();
306            JavaCodeHandler sch = statistics.createStatisticsCodeHandler(ch);
307    
308            directoryParser.initialize(sch, model.getObjectsTable() );
309            directoryParser.parseDirectory(libName, new File(path) );
310    
311            directoryParser.dispose();
312            ch = null;
313    
314            return statistics;
315        }

317        /**
318         * Serializes the content of the JRE {@link com.asemantics.rdfcoder.sourceparse.ObjectsTable}
319         * into the repository.
320         *
321         * @param objectsTable
322         */
323        private void serializeJREObjectsTable(String jreName, ObjectsTable objectsTable)
324        throws RepositoryException, IOException {
325            Repository.Resource resource = repository.createResource(
326                    getJREObjectTableResourceName( jreName ),
327                    Repository.ResourceType.BINARY
328            );
329            OutputStream os = resource.getOutputStream();
330            ObjectOutputStream oos = null;
331            try {
332                oos = new ObjectOutputStream(os);
333                oos.writeObject( objectsTable );
334            } finally {
335                if(oos != null) { oos.close(); }
336            }
337        }
338    
339        /**
340         * Deserializes the content of the JRE {@link com.asemantics.rdfcoder.sourceparse.ObjectsTable}
341         * from the repository.
342         * 
343         * @param jreName
344         * @return
345         * @throws RepositoryException
346         * @throws IOException
347         * @throws ClassNotFoundException
348         */
349        private ObjectsTable deserializeJREObjectsTable(String jreName)
350        throws RepositoryException, IOException, ClassNotFoundException {
351            Repository.Resource resource = repository.getResource( getJREObjectTableResourceName( jreName ) );
352            InputStream is = resource.getInputStream();
353            ObjectInputStream ois = null;
354            try {
355                ois = new ObjectInputStream(is);
356                ObjectsTable ot = (ObjectsTable) ois.readObject();
357                return ot;
358            } finally {
359                if(ois != null) { ois.close(); }
360            }
361        }
