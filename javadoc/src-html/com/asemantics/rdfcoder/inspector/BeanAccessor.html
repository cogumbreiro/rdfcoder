018    package com.asemantics.rdfcoder.inspector;
019    
020    
021    import java.beans.BeanInfo;
022    import java.beans.Introspector;
023    import java.beans.PropertyDescriptor;
024    import java.lang.reflect.Method;
025    
026    /**
027     * Utility class to handle bean properties.
028     */
029    public class BeanAccessor {
030    
031        /**
032         * Returns the property value for the specified <i>bean</i>.
033         *
034         * @param bean
035         * @param propertyName
036         * @return the property of the bean.
037         */
038        public static Object getProperty(Object bean, String propertyName) throws PatternException {
039            if(bean == null) {
040                throw new NullPointerException();
041            }
042            try {
043                Method method = null;
044                String[] candidateMethodNames = getMethodGetterName(propertyName);
045                for(String candidateMethodName : candidateMethodNames) {
046                    try {
047                        method = bean.getClass().getMethod( candidateMethodName );
048                        return method.invoke( bean );
049                    } catch (NoSuchMethodException e) {}
050                }
051                throw new PatternException(
052                        "Cannot find property '" + propertyName + "' on bean '" + bean + "'"
053                );
054            } catch (Exception e) {
055                throw new PatternException(
056                        "Error while retrieving property " + propertyName + "' from bean '" + bean + "'",
057                        e
058                );
059            }
060        }
061    
062        /**
063         * Describes a given <i>bean</i>.
064         * 
065         * @param bean
066         * @return the bean description of the bean.
067         * @throws PatternException
068         */
069        public static String describeBean(Object bean) throws PatternException {
070            if( bean == null ) {
071                throw new NullPointerException("Invalid null bean");
072            }
073    
074            try {
075                BeanInfo beanInfo = Introspector.getBeanInfo( bean.getClass() );
076                PropertyDescriptor[] propertyDescriptors = beanInfo.getPropertyDescriptors();
077                StringBuilder sb = new StringBuilder();
078                sb.append("{\n");
079                for(PropertyDescriptor propertyDescriptor : propertyDescriptors) {
080                    sb.append(propertyDescriptor.getName())
081                            .append(":")
082                    .append( propertyDescriptor.getPropertyType().getName() )
083                            .append("\n");
084                }
085                sb.append("}\n");
086                return sb.toString();
087            } catch (Exception e) {
088                throw new PatternException("Error while inspecting bean: '" + bean + "'", e);
089            }
090        }
091    
092        /**
093         * Returns the expected name for the getter method of given <i>propertyName</i>.
094         *
095         * @param propertyName name of a bean property.
096         * @return the name of a method.
097         */
098        private static String[] getMethodGetterName(String propertyName) {
099            final String methodName = propertyName.substring(0,1);
100            final String postfix = methodName.toUpperCase() + propertyName.substring(1);;
101            return new String[] { "get" + postfix, "is" + postfix, methodName};
102        }
103    }
