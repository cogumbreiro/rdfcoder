027    /**
028     * Contains base methods common to every <i>Code Model</i>.
029     */
030    public abstract class CodeModelBase implements CodeModel, BackTrackingSupport {

032

073        /**
074         * URI-fier method.
075         *
076         * @param prop
077         * @return the URI form as string.

079        protected static String toURI(String prop) {
080            return CODER_URI + prop;
081        }

083        /**
084         * Prefix-fier method.
085         * @param pref
086         * @return the prefix.
087         */
088        protected static final String toPrefix(String pref) {
089            return pref + PREFIX_SEPARATOR;
090        }
091    
092        /**
093         * Unique identifier discriminator counter.
094         */
095        private long _counter = 0;
096    
097        /**
098         * Generates a temporary unique identifier.
099         */
100        public Identifier generateTempUniqueIdentifier() {
101            return IdentifierBuilder
102                    .create()
103                    .pushFragment(System.currentTimeMillis() + "_" + _counter++, TEMPORARY_TYPE_ID_PREFIX)
104                    .build();
105        }
106    
107        /**
108         * Prefixes a full qualifier with the given prefix validating both first.
109         *
110         * @param prefix
111         * @param path
112         * @return the prefixed string.
113         */
114        public static String prefixFullyQualifiedName(String prefix, String path) {
115            if(prefix == null || prefix.trim().length() == 0 || path == null) {
116                throw new IllegalArgumentException();
117            }
118    
119            int prefixSeparatorIndex = path.indexOf(PREFIX_SEPARATOR);
120            // Invalid path: prefix is ""
121            if(prefixSeparatorIndex == 0) {
122                throw new IllegalArgumentException("Invalid path: '" + path + "'");
123            }
124            // Removes the last prefix.
125            if(prefixSeparatorIndex > 0) {
126                path = path.substring(prefixSeparatorIndex + 1);
127            }
128    
129            return prefix + path;
130        }
131    
132        /**
133         * Replaces temporary identifiers with final types.
134         *
135         * @param identifier
136         * @param qualifiedType
137         * @return the number of replaced identifiers.
138         * @see #generateTempUniqueIdentifier()
139         */
140        public int replaceIdentifierWithQualifiedType(final Identifier identifier, final Identifier qualifiedType) {
141            int effectedTriples;
142    
143            /*
144            if( RDFCoder.assertions() && identifier.indexOf(CodeModel.PREFIX_SEPARATOR) == -1) {
145                throw new IllegalArgumentException("identifier: " + identifier + " qualified type: " + qualifiedType);
146            }
147            */
148    
149            final String identifierStr = identifier.getIdentifier();
150    
151            List newTriples = new ArrayList();
152            String[] nextTriple;
153    
154            // Replacing all subjects.
155    
156            // Creates new triples image.
157            TripleIterator ti = searchTriples(identifierStr, CodeModel.ALL_MATCH, JavaCodeModel.ALL_MATCH);
158            while(ti.next()) {
159                newTriples.add( new String[] { ti.getPredicate(), ti.getObject() } );
160            }
161            effectedTriples = newTriples.size();
162    
163            // Delete old triples.
164            Iterator<String[]> newTriplesIter = newTriples.iterator();
165            while(newTriplesIter.hasNext()) {
166                nextTriple = newTriplesIter.next();
167                removeTriple( identifierStr, nextTriple[0], nextTriple[1] );
168            }
169            // Add new ones.
170            newTriplesIter = newTriples.iterator();
171            while(newTriplesIter.hasNext()) {
172                nextTriple = newTriplesIter.next();
173                removeTriple( identifierStr, nextTriple[0], nextTriple[1] );
174            }
175    
176            // Replacing all objects.
177    
178            newTriples.clear();
179    
180            // Creates new triples image.
181            ti = searchTriples(CodeModel.ALL_MATCH, CodeModel.ALL_MATCH, identifierStr);
182            while(ti.next()) {
183                newTriples.add( new String[] {ti.getSubject(), ti.getPredicate() } );
184            }
185            effectedTriples += newTriples.size();
186    
187            // Delete old triples.
188            newTriplesIter = newTriples.iterator();
189            while(newTriplesIter.hasNext()) {
190                nextTriple = newTriplesIter.next();
191                removeTriple( nextTriple[0], nextTriple[1], identifierStr );
192            }
193            // Add new ones.
194            newTriplesIter = newTriples.iterator();
195            while(newTriplesIter.hasNext()) {
196                nextTriple = newTriplesIter.next();
197                removeTriple( nextTriple[0], nextTriple[1], qualifiedType.getIdentifier() );
198            }
199    
200            return effectedTriples;
201        }
202    
203        /**
204         * Returns the prefix associated to a <i>RDF</i> type.
205         *
206         * @param rdfType
207         * @return the prefix string.
208         */
209        public static String getPrefixFromRDFType(String rdfType) {
210            if(rdfType.indexOf(CODER_URI) < 0) {
211                throw new IllegalArgumentException("Invalid rdfType: '" + rdfType + "'");
212            }
213            return rdfType.substring(CODER_URI.length()) + CodeModel.PREFIX_SEPARATOR;
214        }
215    
216        /**
217         * Returns <code>true</code> if the path is already prefixed, <code>false</code>
218         * otherwise.
219         *
220         * @param path the path to check.
221         * @return <code>true</code> if the path is prefixed.
222         */
223        protected static boolean isPrefixed(String path) {
224             return path.indexOf(CodeModel.PREFIX_SEPARATOR) != -1;
225        }
226    
227        /**
228         * Prefixes a parameter with the given prefix validating both first.
229         *
230         * @param prefix
231         * @param parameter
232         * @return the prefixed string.
233         */
234        // TODO: HIGH : remove it.
235        public static String prefixParameter(String prefix, String parameter) {
236            if(prefix == null || prefix.trim().length() == 0 || parameter == null || parameter.trim().length() == 0) {
237                throw new IllegalArgumentException();
238            }
239    
240            int prefLocation = parameter.indexOf(prefix);
241            if(prefLocation == -1) { // Not pefixed.
242                return prefix + parameter;
243            } else if(prefLocation == 0) { // Already prefixed.
244                return parameter;
245            } else {
246                throw new IllegalArgumentException(
247                    "Something wrong in parameter '" + parameter + "' in applying prefix '" + prefix + "'"
248                );
249            }
250        }
