019    package com.asemantics.rdfcoder.model;
020    
021    import com.asemantics.rdfcoder.model.java.JavaCodeHandler;
022    import com.asemantics.rdfcoder.model.java.JavaCodeModel;
023    
024    import java.util.ArrayList;
025    import java.util.List;
026    
027    /**
028     * This class provides facility methods to read a <i>context</i>
029     * built using the {@link com.asemantics.rdfcoder.model.IdentifierBuilder}.
030     *
031     * @see com.asemantics.rdfcoder.model.IdentifierBuilder
032     * @version $Id$
033     */
034    public class IdentifierReader {
035    
036        /**
037         * Reads an identifier generated by the {@link com.asemantics.rdfcoder.model.IdentifierBuilder}.
038         *
039         * @param identifier
040         * @return the parsed identifier.
041         * @throws IllegalArgumentException if an error occurred during parsing.
042         * @see com.asemantics.rdfcoder.model.IdentifierBuilder
043         */
044        public static Identifier readIdentifier(String identifier) {
045            final int endPrefixIndex = identifier.indexOf(CodeModel.URI_PREFIX_SEPARATOR);
046            final String prefix = identifier.substring(0, endPrefixIndex + 1);
047            final String postfix = identifier.substring(endPrefixIndex + 1);
048            final String[] sections = postfix.split("\\" + JavaCodeHandler.PACKAGE_SEPARATOR);
049            IdentifierFragment[] fragments = new IdentifierFragment[sections.length];
050            int i = 0;
051            String[]  sectionSplit;
052            String currentQualifier = null;
053            for(String section : sections) {
054                sectionSplit = section.split(CodeModel.PREFIX_SEPARATOR);
055                if(sectionSplit.length > 2) {
056                    throw new IllegalArgumentException(
057                            String.format("Invalid section [%s] in identifier [%s] ", section, identifier)
058                    );
059                }
060                if(sectionSplit.length == 2) {
061                    currentQualifier = sectionSplit[0];
062                    if(currentQualifier.length() == 0) {
063                        throw new IllegalArgumentException(
064                            String.format("Invalid 0 length section qualifier in identifier [%s]", identifier)
065                        );
066                    }
067                } else {
068                    if(currentQualifier == null) {
069                        throw new IllegalArgumentException(
070                            String.format("Invalid identifier [%s]: must start with a qualifier.", identifier)
071                        );
072                    }
073                }
074                fragments[i] = new IdentifierFragment(
075                        sectionSplit.length == 2 ? sectionSplit[1] : sectionSplit[0],
076                        currentQualifier
077                );
078                i++;
079            }
080            return new Identifier(prefix, fragments);
081        }
082    
083        /**
084         * Reads a package and returns the representing identifier.
085         *
086         * @param pack the package to be read.
087         * @return the parsed identifier.
088         * @throws IllegalArgumentException if an error occurred during parsing.
089         */
090        public static Identifier readPackage(String pack) {
091            if(pack.contains(CodeModel.PREFIX_SEPARATOR)) {
092                throw new IllegalArgumentException("Invalid package: " + pack);
093            }
094            String[] sections = pack.split("\\" + JavaCodeHandler.PACKAGE_SEPARATOR);
095            IdentifierFragment[] fragments = new IdentifierFragment[sections.length];
096            int i = 0;
097            for(String section : sections) {
098                fragments[i] = new IdentifierFragment(section, JavaCodeModel.PACKAGE_KEY);
099                i++;
100            }
101            return new Identifier("", fragments);
102        }
103    
104        /**
105         * Reads a fully qualified class.
106         *
107         * @param clazz the class name.
108         * @return the identifier representing the class.
109         * @throws IllegalArgumentException if an error occurred during parsing.
110         */
111        public static Identifier readFullyQualifiedClass(String clazz) {
112            int packageEnd = clazz.lastIndexOf(JavaCodeHandler.PACKAGE_SEPARATOR);
113            List<IdentifierFragment> fragments = new ArrayList<IdentifierFragment>();
114            if(packageEnd == -1) {
115                fragments.add( new IdentifierFragment("", JavaCodeModel.PACKAGE_KEY) );
116            } else {
117                Identifier packageIdentifier = IdentifierReader.readPackage(clazz.substring(0, packageEnd));
118                fragments.addAll( packageIdentifier.fragments );
119            }
120            fragments.add( new IdentifierFragment( clazz.substring(packageEnd + 1), JavaCodeModel.CLASS_KEY ) );
121            return new Identifier("", fragments);
122        }
123    
124    }
