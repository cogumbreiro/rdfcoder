054        public void parse(JavaBytecodeClassLoader classloader, InputStream inputStream, String fileName) throws IOException {
055            containersStack.clear();
056    
057            // Parse bytecode.
058            ClassParser classParser = new ClassParser( inputStream, fileName );
059            JavaClass javaClass = classParser.parse();
060    
061            final CodeHandler codeHandler   = (CodeHandler) getParseHandler();
062            ObjectsTable objectsTable = getObjectsTable();
063    
064            // Package.
065            codeHandler.startPackage( javaClass.getPackageName() );
066            try {
067    
068                // Objects table update.
069                objectsTable.addObject( javaClass.getClassName() );
070    
071                // Recursive processing.
072                Set processed = new HashSet();
073                processClass(codeHandler, classloader, processed, javaClass);
074    
075            } finally {
076                // Close package.
077                codeHandler.endPackage();
078            }
079        }
080    
081        public void parse(JavaBytecodeClassLoader classloader, File file) throws IOException {
082            if(file == null) {
083                throw new NullPointerException();
084            }
085            if( ! file.exists() ) {
086                throw new IllegalArgumentException("cannot find file: " + file.getAbsolutePath());
087            }
088    
089            InputStream inputStream = new FileInputStream(file);
090            parse(classloader, inputStream, file.getAbsolutePath());
091            inputStream.close();
092        }

094        public void parse(File file) throws IOException {
095            //TODO - HIGH: check this method: JavaBytecodeClassLoader must not be null.
096            parse(null, file);
097        }

099        protected void processClass(CodeHandler codeHandler, JavaBytecodeClassLoader classloader, Set processed, JavaClass javaClass ) {
100    
101            containersStack.push( javaClass );
102    
103            codeHandler.startCompilationUnit(javaClass.toString());
104    
105            // The stack is currectly handled despite errors.
106            try {
107    
108                Field[] fields = javaClass.getFields();
109                List<String> enumElements = new ArrayList<String>();
110    
111                // Extracts class modifiers.
112                JavaCodeModel.JModifier[] modifiers = extractModifiers(javaClass);
113    
114                if( javaClass.isEnum() ) { // Enumeration.
115                    //TODO: MED - Check this solution.
116                    for(int i = 0; i < fields.length - 1; i++) {   // Skip last element containing $VALUE
117                        if( fields[i].getType().toString().equals( javaClass.getClassName() ) ) { // Excluding real enumElements.
118                            enumElements.add( fields[i].getName() );
119                        }
120                    }
121                    codeHandler.startEnumeration(
122                            modifiers,
123                            toVisibility( javaClass ),
124                            toQualifiedClassName( javaClass ),
125                            enumElements.toArray( new String[ enumElements.size() ] )
126                    );
127                } else if( javaClass.isClass() ) { // Class.
128                    codeHandler.startClass(
129                       modifiers,
130                       toVisibility(javaClass),
131                       toQualifiedClassName(javaClass),
132                       toQualifiedSuperClassName(javaClass),
133                       toQualifiedInterfaces(javaClass)
134                    );
135                } else if( javaClass.isInterface() ) { // Interface.
136                    codeHandler.startInterface(
137                       toQualifiedClassName(javaClass),
138                       toQualifiedInterfaces(javaClass)
139                    );
140                } else {
141                    throw new IllegalStateException();
142                }
143    
144                // Fields.
145                for(int i = 0; i < fields.length; i++) {
146                    if( enumElements.contains( fields[i].getName() ) ) {
147                        continue;
148                    }
149                    codeHandler.attribute(
150                            extractModifiers(fields[i]),
151                            toVisibility(fields[i]),
152                            toQualifiedAttribute(fields[i]),
153                            bcelTypeToJType( fields[i].getType() ),
154                            toConstantValue( fields[i].getConstantValue() )
155                    );
156                }
157                enumElements.clear();
158    
159    
160                // Methods.
161                Method[] methods = javaClass.getMethods();
162                for(int i = 0; i < methods.length; i++) {
163    
164                    if( INIT_METHOD.equals(methods[i].getName()) ) { // Constructors.
165                        codeHandler.constructor(
166                            extractModifiers(methods[i]),
167                            toVisibility( methods[i] ),
168                            i,
169                            toParameterNames( methods[i].getArgumentTypes()  ),
170                            toParameterTypes( methods[i].getArgumentTypes()  ),
171                            toExceptionTypes( methods[i].getExceptionTable() )
172                        );
173                    }
174    
175                    codeHandler.method(
176                            extractModifiers(methods[i]),
177                            toVisibility( methods[i] ),
178                            toQualifiedMethod( methods[i] ),
179                            i,
180                            toParameterNames( methods[i].getArgumentTypes() ),
181                            toParameterTypes( methods[i].getArgumentTypes() ),
182                            bcelTypeToJType ( methods[i].getReturnType()    ),
183                            toExceptionTypes( methods[i].getExceptionTable())
184                    );
185                }
186    
187                // Inner classes. If class loader is not defined the inner classes are ignored.
188                if(classloader != null) {
189                    String innerClassName = null;
190                    InnerClass[] innerClasses = innerClasses(javaClass);
191                    ConstantPool constantPool = javaClass.getConstantPool();
192                    for(int i = 0; i < innerClasses.length; i++) {
193                        innerClassName = constantPool.getConstantString(
194                            innerClasses[i].getInnerClassIndex(),
195                            Constants.CONSTANT_Class
196                        );
197                        if( processed.contains(innerClassName) ) {
198                            continue;
199                        } else {
200                            processed.add(innerClassName);
201                        }
202                        System.out.println("INNER CLASS: " + innerClassName);
203                        innerClassName = Utility.compactClassName(innerClassName);
204                        JavaClass innerClass = null;
205                        try {
206                            innerClass = classloader.loadClass(innerClassName);
207                            if(innerClass == null) {
208                                System.out.println("CANNOT LOAD class: " + innerClassName);
209                                continue;
210                            }
211                        } catch (IOException ioe) {
212                            codeHandler.parseError(javaClass.getClassName(), "[" + ioe.getClass().getName() + "]: " + ioe.getMessage());
213                            ioe.printStackTrace();
214                            continue;
215                        }
216                    }
217                }
218    
219             } finally {
220    
221                // End container.
222                if( javaClass.isClass() ) {
223                    codeHandler.endClass();
224                } else if( javaClass.isInterface() ) {
225                    codeHandler.endInterface();
226                } else if(javaClass.isEnum()) {
227                    codeHandler.endEnumeration();
228                } else {
229                    throw new IllegalStateException();
230                }
231    
232                containersStack.pop();
233                codeHandler.endCompilationUnit();
234            }
235        }

237        private InnerClass[] innerClasses(JavaClass javaClass) {
238            Attribute[] attributes = javaClass.getAttributes();
239            for(int i = 0; i < attributes.length; i++) {
240                if( attributes[i] instanceof InnerClasses ) {
241                    InnerClasses innerClasses = (InnerClasses) attributes[i];
242                    return innerClasses.getInnerClasses();
243                }
244            }
245            return new InnerClass[0];
246        }

248        private JavaCodeModel.JVisibility toVisibility(AccessFlags af) {
249            if( af.isPublic() ) {
250                return JavaCodeModel.JVisibility.PUBLIC;
251            } else if( af.isProtected() ) {
252                return JavaCodeModel.JVisibility.PROTECTED;
253            } else if( af.isPrivate() ) {
254                return JavaCodeModel.JVisibility.PRIVATE;
255            } else {
256                return JavaCodeModel.JVisibility.DEFAULT;
257            }
258        }

275        private String toQualifiedClassName(JavaClass javaClass) {
276            return getContainerPath() + CodeHandler.PACKAGE_SEPARATOR + javaClass.getClassName();
277        }
278    
279        private String toQualifiedSuperClassName(JavaClass javaClass) {
280            JavaClass superClass;
281            try {
282                superClass = javaClass.getSuperClass();
283            } catch (ClassNotFoundException e) {
284                return null;
285            }
286    
287            return superClass.getPackageName() + CodeHandler.PACKAGE_SEPARATOR + superClass.getClassName();
288        }
289    
290        private String[] toQualifiedInterfaces(JavaClass javaClass) {
291            JavaClass[] interfaces;
292            try {
293                interfaces = javaClass.getAllInterfaces();
294            } catch (ClassNotFoundException e) {
295                return null;
296            }
297            String[] result = new String[interfaces.length];
298            for(int i = 0; i < interfaces.length; i++) {
299                result[i] = interfaces[i].getPackageName() + CodeHandler.PACKAGE_SEPARATOR + interfaces[i].getClassName();
300            }
301            return result;
302        }
303    
304        private String toQualifiedAttribute(Field field) {
305            return getContainerPath() + CodeHandler.PACKAGE_SEPARATOR + field.getName();
306        }
307    
308        private String toQualifiedMethod(Method method) {
309            return getContainerPath() + CodeHandler.PACKAGE_SEPARATOR + method.getName();
310        }

312        private String toConstantValue(ConstantValue cv) {
313            if(cv == null) {
314                return null;
315            }
316            ConstantPool cp =  cv.getConstantPool();
317            Constant c = cp.getConstant( cv.getConstantValueIndex() );
318            return c.toString();
319        }

321        private JavaCodeModel.JType bcelTypeToJType(org.apache.bcel.generic.Type type) {
322    
323            // Convert basic types.
324            if(type == org.apache.bcel.generic.Type.VOID ) {
325                 return JavaCodeModel.VOID;
326            } else if( type == org.apache.bcel.generic.Type.BOOLEAN ) {
327                return JavaCodeModel.BOOL;
328            } else if( type == org.apache.bcel.generic.Type.INT ) {
329                return JavaCodeModel.INT;
330            } else if( type == org.apache.bcel.generic.Type.SHORT ) {
331                return JavaCodeModel.SHORT;
332            } else if( type == org.apache.bcel.generic.Type.BYTE ) {
333                return JavaCodeModel.BYTE;
334            } else if( type == org.apache.bcel.generic.Type.LONG ) {
335                return JavaCodeModel.LONG;
336            } else if( type == org.apache.bcel.generic.Type.DOUBLE) {
337                return JavaCodeModel.DOUBLE;
338            } else if( type == org.apache.bcel.generic.Type.FLOAT ) {
339                return JavaCodeModel.FLOAT;
340            } else if( type == org.apache.bcel.generic.Type.CHAR ) {
341                return JavaCodeModel.CHAR;
342            }
343    
344            // Convert object types.
345            if( type instanceof org.apache.bcel.generic.ObjectType) {
346                org.apache.bcel.generic.ObjectType bcelOT = (org.apache.bcel.generic.ObjectType) type;
347                return new JavaCodeModel.ObjectType( bcelOT.getClassName() );
348            }
349    
350            // Array types.
351            if ( type instanceof org.apache.bcel.generic.ArrayType ) {
352                org.apache.bcel.generic.ArrayType arrayType = (org.apache.bcel.generic.ArrayType) type;
353                return new JavaCodeModel.ArrayType(
354                        bcelTypeToJType(arrayType.getBasicType() ),
355                        arrayType.getDimensions()
356                );
357            }
358    
359            throw new IllegalArgumentException("cannot convert " + type);
360    
361        }

363        private String toParameterName(org.apache.bcel.generic.Type type ) {
364            // Generate name for basic type.
365            if( type == org.apache.bcel.generic.Type.BOOLEAN ) {
366                return "b";
367            } else if( type == org.apache.bcel.generic.Type.INT ) {
368                return "i";
369            } else if( type == org.apache.bcel.generic.Type.SHORT ) {
370                return "s";
371            } else if( type == org.apache.bcel.generic.Type.BYTE ) {
372                return "e";
373            } else if( type == org.apache.bcel.generic.Type.LONG ) {
374                return "l";
375            } else if( type == org.apache.bcel.generic.Type.DOUBLE) {
376                return "d";
377            } else if( type == org.apache.bcel.generic.Type.FLOAT ) {
378                return "f";
379            } else if( type == org.apache.bcel.generic.Type.CHAR ) {
380                return "c";
381            }
382    
383            // Generate name for object type.
384            if( type instanceof org.apache.bcel.generic.ObjectType) {
385                org.apache.bcel.generic.ObjectType bcelOT = (org.apache.bcel.generic.ObjectType) type;
386                String cn = bcelOT.getClassName();
387                int i = cn.lastIndexOf(".");
388                String result = cn.substring(i + 1);
389                if("String".equals(result)) {
390                    return "s";
391                } else if("Object".equals(result)) {
392                    return "o";
393                } else {
394                    return result;
395                }
396            }
397    
398            // Generate name for array type.
399            if ( type instanceof org.apache.bcel.generic.ArrayType ) {
400                org.apache.bcel.generic.ArrayType arrayType = (org.apache.bcel.generic.ArrayType) type;
401                String typeName = toParameterName( arrayType.getBasicType() );
402                return typeName + "s";
403            }
404    
405            throw new IllegalArgumentException("cannot generate name for " + type);
406        }

409        private String[] toParameterNames( org.apache.bcel.generic.Type[] type ) {
410            HashMap<String,Integer> namesMap = new HashMap();
411            String[] names = new String[type.length];
412            String designedName;
413            for(int i = 0; i < type.length; i++) {
414                designedName = toParameterName( type[i] ).toLowerCase();
415                Integer occurrences = namesMap.get(designedName);
416                if(occurrences == null) { occurrences = new Integer(0); }
417                occurrences++;
418                namesMap.put(designedName, occurrences);
419                names[i] = designedName + ( occurrences == 1 ? "" : occurrences );
420            }
421            namesMap.clear();
422            return names;
423        }

425        private JavaCodeModel.JType[] toParameterTypes( org.apache.bcel.generic.Type[] types ) {
426            JavaCodeModel.JType[] result = new JavaCodeModel.JType[types.length];
427            for(int i = 0; i < types.length; i++) {
428                result[i] = bcelTypeToJType(types[i]);
429            }
430            return result;
431        }

433        private JavaCodeModel.ExceptionType[] toExceptionTypes(ExceptionTable exceptionTable) {
434            if(exceptionTable == null) {
435                return null;
436            }
437            String[] exceptionNames = exceptionTable.getExceptionNames();
438            JavaCodeModel.ExceptionType[] exceptions = new JavaCodeModel.ExceptionType[exceptionNames.length];
439            for(int i = 0; i < exceptionNames.length; i++) {
440                exceptions[i] = new JavaCodeModel.ExceptionType(exceptionNames[i]);
441            }
442            return exceptions;
443        }

445        /**
446         * Extracts the list of modifiers associated to the given entity.
447         * @param accessFlags
448         * @return
449         */
450        private JavaCodeModel.JModifier[] extractModifiers(AccessFlags accessFlags) {
451            List<JavaCodeModel.JModifier> modifiers = new ArrayList<JavaCodeModel.JModifier>(JavaCodeModel.JModifier.values().length);
452            if( accessFlags.isAbstract() ) {
453                modifiers.add(JavaCodeModel.JModifier.ABSTRACT);
454            }
455            if( accessFlags.isFinal() ) {
456                modifiers.add(JavaCodeModel.JModifier.FINAL);
457            }
458            if( accessFlags.isStatic() ) {
459                modifiers.add(JavaCodeModel.JModifier.STATIC);
460            }
461            if( accessFlags.isVolatile() ) {
462                modifiers.add(JavaCodeModel.JModifier.VOLATILE);
463            }
464            if( accessFlags.isNative() ) {
465                modifiers.add(JavaCodeModel.JModifier.NATIVE);
466            }
467            if( accessFlags.isTransient() ) {
468                modifiers.add(JavaCodeModel.JModifier.TRANSIENT);
469            }
470            if( accessFlags.isSynchronized() ) {
471                modifiers.add(JavaCodeModel.JModifier.SYNCHRONIZED);
472            }
473            return modifiers.toArray(new JavaCodeModel.JModifier[modifiers.size()]);
474        }
