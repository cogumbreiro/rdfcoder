104        /**
105         * Processes a single compilation unit.
106         *
107         * @param location
108         * @param ast
109         */
110        public void processCompilationUnit(String location, ASTCompilationUnit ast) {
111    
112            codeHandler = (CodeHandler) getParseHandler();
113    
114            try {
115                codeHandler.startCompilationUnit(location);
116            } catch (Throwable t) {
117                t.printStackTrace();
118            }
119    
120            // Retrieve package.
121            List packs = ast.findChildrenOfType(ASTPackageDeclaration.class);
122            String packagePath;
123            if (packs.size() > 0) {
124                packagePath = ((ASTName) ((ASTPackageDeclaration) packs.get(0)).findChildrenOfType(ASTName.class).get(0)).getName();
125            } else {
126                packagePath = ""; // Default package.
127            }
128            try {
129                codeHandler.startPackage(packagePath);
130            } catch (Throwable t) {
131                t.printStackTrace();
132            }
133    
134            try {
135    
136                // Define inports context.
137                ImportsContext importsContext = new ImportsContext();
138                importsContext.setContextPackage(packagePath);
139    
140                // Retrieve imports.
141                List imports = ast.findChildrenOfType(ASTImportDeclaration.class);
142                for (int i = 0; i < imports.size(); i++) {
143                    ASTImportDeclaration importDeclaration = (ASTImportDeclaration) imports.get(i);
144                    String importEntry = ((ASTName) importDeclaration.findChildrenOfType(ASTName.class).get(0)).getName();
145                    if (importDeclaration.isImportOnDemand()) { //Starred package.
146                        try {
147                            importsContext.addStarredPackage(importEntry);
148                        } catch(IllegalArgumentException iae) {
149                            codeHandler.parseError( location, iae.getClass().getName()  + "[" + iae.getMessage() + "]" );
150                        }
151                    } else {
152                        importsContext.addFullyQualifiedObject(importEntry);
153                    }
154                }
155    
156                // Note: enumerations can be present only at first level.
157                extractEnumerations(
158                        packagePath,
159                        codeHandler,
160                        (ASTCompilationUnit) ast.findChildrenOfType(ASTCompilationUnit.class).get(0)
161                );
162    
163                // Process first level entities.
164                processLevel(packagePath, importsContext, ast);
165    
166            } finally {
167                try {
168                    codeHandler.endPackage();
169                }catch (Throwable t) {
170                    t.printStackTrace();
171                }
172                try {
173                    codeHandler.endCompilationUnit();
174                } catch (Throwable t) {
175                    t.printStackTrace();
176                }
177            }
178        }

180        private void processLevel(String packagePath, ImportsContext importsContext, SimpleNode simpleNode) {
181    
182            // Find classes.
183            List<ASTClassDeclaration> classes = findChildrenAtSameLevel(simpleNode, ASTClassDeclaration.class);
184    
185            // Find interfaces.
186            List<ASTInterfaceDeclaration> interfaces = findChildrenAtSameLevel(simpleNode, ASTInterfaceDeclaration.class);
187    
188            // Process Class.
189            String className;
190            String superClass = null;
191            for (ASTClassDeclaration classDeclaration : classes ) {
192    
193                List<ASTUnmodifiedClassDeclaration> unmodifiedClassDeclarations = classDeclaration.findChildrenOfType(ASTUnmodifiedClassDeclaration.class);
194                for( NamedNode unmodifiedClassDeclaration : unmodifiedClassDeclarations) {
195    
196                    // Class name.
197                    className = unmodifiedClassDeclaration.getName(); // Class name.
198    
199                    getObjectsTable().addObject(packagePath, className);
200    
201                    // Class visibility.
202                    JavaCodeModel.JVisibility clsOrIntVisibility = retrieveVisibility(classDeclaration);
203    
204                    // Find super class.
205                    List classOrInterface = unmodifiedClassDeclaration.findChildrenOfType(ASTClassOrInterfaceType.class);
206                    if (classOrInterface.size() > 0) {   // Has super Class.
207                        superClass = retrieveObjectName( ((ASTClassOrInterfaceType) classOrInterface.get(0)).findChildrenOfType(ASTIdentifier.class) );
208                    }
209    
210                    codeHandler.startClass(
211                            //TODO: verify the subsequest row.
212                            extractModifiers( classDeclaration ),
213                            clsOrIntVisibility,
214                            packagePath + CodeHandler.PACKAGE_SEPARATOR + className,
215                            superClass != null ? qualifyType(importsContext, superClass, new JavaCodeModel.ObjectType(null) ) : null,
216                            extractImplementedInterfaces(importsContext, unmodifiedClassDeclaration)
217                    );
218                    try {
219                        extractEnumerations(packagePath, codeHandler, unmodifiedClassDeclaration);
220                        extractAttributes  (packagePath, codeHandler, importsContext, unmodifiedClassDeclaration);
221                        extractContructors (codeHandler, importsContext, unmodifiedClassDeclaration);
222                        extractMethods     (packagePath, codeHandler, importsContext, unmodifiedClassDeclaration);
223    
224                        // Recursive invocation.
225                        processLevel(packagePath + CodeHandler.PACKAGE_SEPARATOR + className, importsContext, unmodifiedClassDeclaration);
226    
227                    } finally {
228                        codeHandler.endClass();
229                    }
230    
231                }
232            }
233    
234            // Process interface.
235            String interfaceName = null;
236            for(ASTInterfaceDeclaration interfaceDeclaration : interfaces ) {
237    
238                List<ASTUnmodifiedInterfaceDeclaration> unmodifiedInterfaceDeclarations = interfaceDeclaration.findChildrenOfType(ASTUnmodifiedInterfaceDeclaration.class);
239                for(NamedNode unmodifiedInterfaceDeclaration : unmodifiedInterfaceDeclarations) {
240    
241                    // Interface name.
242                    interfaceName = unmodifiedInterfaceDeclaration.getName();
243    
244                    getObjectsTable().addObject(packagePath, interfaceName);
245    
246                    codeHandler.startInterface(
247                            packagePath + CodeHandler.PACKAGE_SEPARATOR + interfaceName,
248                            extractImplementedInterfaces(importsContext, unmodifiedInterfaceDeclaration)
249                    );
250                    try {
251                        extractEnumerations(packagePath, codeHandler, unmodifiedInterfaceDeclaration);
252                        extractAttributes  (packagePath, codeHandler, importsContext, unmodifiedInterfaceDeclaration);
253                        extractMethods     (packagePath, codeHandler, importsContext, unmodifiedInterfaceDeclaration);
254    
255                        // Recursive invocation.
256                        processLevel(packagePath + CodeHandler.PACKAGE_SEPARATOR + interfaceName, importsContext, unmodifiedInterfaceDeclaration);
257    
258                    } finally {
259                        codeHandler.endInterface();
260                    }
261    
262                }
263            }
264    
265        }

267        /**
268         * Extracts the list of modifiers associated to this access node.
269         * @param accessNode
270         * @return
271         */
272        private JavaCodeModel.JModifier[] extractModifiers(AccessNode accessNode) {
273            List<JavaCodeModel.JModifier> modifiers = new ArrayList<JavaCodeModel.JModifier>(JavaCodeModel.JModifier.values().length);
274            if( accessNode.isAbstract() ) {
275                modifiers.add(JavaCodeModel.JModifier.ABSTRACT);
276            }
277            if( accessNode.isFinal() ) {
278                modifiers.add(JavaCodeModel.JModifier.FINAL);
279            }
280            if( accessNode.isStatic() ) {
281                modifiers.add(JavaCodeModel.JModifier.STATIC);
282            }
283            if( accessNode.isVolatile() ) {
284                modifiers.add(JavaCodeModel.JModifier.VOLATILE);
285            }
286            if( accessNode.isNative() ) {
287                modifiers.add(JavaCodeModel.JModifier.NATIVE);
288            }
289            if( accessNode.isTransient() ) {
290                modifiers.add(JavaCodeModel.JModifier.TRANSIENT);
291            }
292            if( accessNode.isSynchronized() ) {
293                modifiers.add(JavaCodeModel.JModifier.SYNCHRONIZED);
294            }
295            return modifiers.toArray(new JavaCodeModel.JModifier[modifiers.size()]);
296        }

300         * Returns an array containing the fully quelified types of the implemented interfaces
301         * of the current simple node.
302         *
303         * @param importsContext
304         * @param simpleNode
305         * @return
306         */
307        private String[] extractImplementedInterfaces(ImportsContext importsContext, SimpleNode simpleNode) {
308            String[] interfacesArray = null;
309            List genericNameList = simpleNode.findChildrenOfType(ASTGenericNameList.class);
310            for(int i = 0; i < genericNameList.size(); i++) {
311                List classOrInterfaceTypes = ( (ASTGenericNameList) genericNameList.get(i) ).findChildrenOfType(ASTClassOrInterfaceType.class);
312                interfacesArray = new String[classOrInterfaceTypes.size()];
313                for(int j = 0; j < classOrInterfaceTypes.size(); j++) {
314                    List identifierList = ( (ASTClassOrInterfaceType) classOrInterfaceTypes.get(j)).findChildrenOfType(ASTIdentifier.class);
315                    interfacesArray[j] = qualifyType(importsContext, retrieveObjectName(identifierList), new JavaCodeModel.InterfaceType(null) );
316                }
317            }
318            return interfacesArray;
319        }

321        private void extractAttributes(String packagePath, CodeHandler ch, ImportsContext importsContext, NamedNode namedNode) {
322            List attributes = namedNode.findChildrenOfType(ASTFieldDeclaration.class);
323            String attributeName;
324            JavaCodeModel.JVisibility attributeVisibility;
325            JavaCodeModel.JType attributeType;
326            for (int f = 0; f < attributes.size(); f++) {
327                ASTFieldDeclaration fieldDeclaration = (ASTFieldDeclaration) attributes.get(f);
328    
329                // Retrieve name.
330                ASTVariableDeclaratorId variableDeclaratorId =
331                        (ASTVariableDeclaratorId) fieldDeclaration.findChildrenOfType(ASTVariableDeclaratorId.class).get(0);
332                attributeName = variableDeclaratorId.getName();
333    
334                // Visibility.
335                attributeVisibility = retrieveVisibility(fieldDeclaration);
336                
337                // Retrieve array size.
338                int arraySize;
339                 boolean isArray;
340                List referenceTypes = fieldDeclaration.findChildrenOfType(ASTReferenceType.class);
341                if( referenceTypes.size() > 0 ) {
342                     ASTReferenceType astReferenceType = (ASTReferenceType) referenceTypes.get(0);
343                     arraySize = astReferenceType.getArrayCount();
344                     isArray = arraySize > 0;
345                } else {
346                    arraySize = 0;
347                    isArray = false;
348                }
349    
350                // Retrieve type.
351                List primitiveType = fieldDeclaration.findChildrenOfType(ASTPrimitiveType.class);
352                if (primitiveType.size() > 0) { // Primitive type declared.
353                    JavaCodeModel.JType type = JavaCodeModel.javaTypeToJType(((ASTPrimitiveType) primitiveType.get(0)).getName());
354                    attributeType =
355                            isArray
356                                    ?
357                            new JavaCodeModel.ArrayType(type, arraySize)
358                                    :
359                            type;
360                } else { // Obj identifier declared.
361                    String typeName = retrieveObjectName( fieldDeclaration.findChildrenOfType(ASTIdentifier.class) );
362                    JavaCodeModel.JType type = new JavaCodeModel.ObjectType(
363                            qualifyType(
364                                    importsContext,
365                                    typeName,
366                                    isArray
367                                            ?
368                                            new JavaCodeModel.ArrayType(
369                                                new JavaCodeModel.ObjectType(null),
370                                                arraySize
371                                            )
372                                            :
373                                            new JavaCodeModel.ObjectType(null)
374                            )
375                    );
376                    attributeType = isArray ? new JavaCodeModel.ArrayType(type, variableDeclaratorId.getArrayCount()) : type;
377                }
378    
379                ch.attribute(
380                        extractModifiers(fieldDeclaration),
381                        attributeVisibility,
382                        qualifyAttribute(packagePath, attributeName), attributeType, null);
383            }
384        }

386        private void extractContructors(CodeHandler ch, ImportsContext importsContext, NamedNode namedNode) {
387            // Constructors.
388            List constructors = namedNode.findChildrenOfType(ASTConstructorDeclaration.class);
389            ASTConstructorDeclaration constructorDeclaration;
390            for(int c = 0; c < constructors.size(); c++) {
391    
392                constructorDeclaration = ((ASTConstructorDeclaration) constructors.get(c));
393                List formalParameters = constructorDeclaration.findChildrenOfType(ASTFormalParameter.class);
394    
395                // Extract parameters.
396                ASTFormalParameter formalParameter;
397                List<String> parameterNames = new ArrayList<String>();
398                List<JavaCodeModel.JType> parameterTypes = new ArrayList<JavaCodeModel.JType>();
399                for( int fp = 0; fp < formalParameters.size(); fp++ ) {
400                    formalParameter = (ASTFormalParameter) formalParameters.get(fp);
401                    parameterNames.add(
402                            ( (ASTVariableDeclaratorId) formalParameter.findChildrenOfType(ASTVariableDeclaratorId.class).get(0)).getName()
403                    );
404                    parameterTypes.add( extractFormalparameterType(importsContext, formalParameter) );
405                }
406    
407                // Extract exceptions.
408                JavaCodeModel.ExceptionType[] exceptions = extractExceptions(importsContext, constructorDeclaration);
409    
410                ch.constructor(
411                        extractModifiers(constructorDeclaration),
412                        retrieveVisibility(constructorDeclaration),
413                        c,
414                        parameterNames.toArray( new String[parameterNames.size()] ),
415                        parameterTypes.toArray( new JavaCodeModel.JType[parameterTypes.size()] ),
416                        exceptions
417                );
418            }
419        }

421        private void extractMethods(String packagePath, CodeHandler ch, ImportsContext importsContext, NamedNode namedNode) {
422            List methods = namedNode.findChildrenOfType(ASTMethodDeclaration.class);
423            String methodName;
424            JavaCodeModel.JVisibility methodVisibility;
425            boolean returnTypeIsArray;
426            int returnTypeArraySize;
427            JavaCodeModel.JType returnType;
428            List<JavaSourceFileParser.Method> methodsBuffer = new ArrayList<JavaSourceFileParser.Method>();
429            for (int m = 0; m < methods.size(); m++) {
430                ASTMethodDeclaration methodDeclaration = (ASTMethodDeclaration) methods.get(m);
431    
432                // Retrieve method name.
433                methodName = ((ASTMethodDeclarator) methodDeclaration.findChildrenOfType(ASTMethodDeclarator.class).get(0)).getName();
434    
435                // Retrieve method visibility.
436                methodVisibility = retrieveVisibility(methodDeclaration);
437    
438                // Return type is array.
439                List referenceType = methodDeclaration.findChildrenOfType(ASTReferenceType.class);
440                if (referenceType.size() > 0) {
441                    returnTypeArraySize = ((ASTReferenceType) referenceType.get(0)).getArrayCount();
442                    returnTypeIsArray = returnTypeArraySize > 0;
443                } else {
444                    returnTypeArraySize = 0;                
445                    returnTypeIsArray = false;
446                }
447    
448    
449                // Find return type.
450                ASTResultType resultType = (ASTResultType) methodDeclaration.findChildrenOfType(ASTResultType.class).get(0);
451                List primitiveType = resultType.findChildrenOfType(ASTPrimitiveType.class);
452                if (primitiveType.size() > 0) { // Result type is primitive.
453                    JavaCodeModel.JType type = JavaCodeModel.javaTypeToJType(((ASTPrimitiveType) primitiveType.get(0)).getName());
454                    if( returnTypeIsArray ) {
455                        returnType = new JavaCodeModel.ArrayType(type, returnTypeArraySize);
456                    } else {
457                       returnType = type;
458                    }
459                } else if (resultType.findChildrenOfType(ASTIdentifier.class).size() > 0) { // Result type is Object.
460                    String qualifiedType = qualifyType(
461                            importsContext,
462                            retrieveObjectName( resultType.findChildrenOfType(ASTIdentifier.class) ),
463                            returnTypeIsArray
464                                    ?
465                                    new JavaCodeModel.ArrayType(
466                                        new JavaCodeModel.ObjectType(null),
467                                        returnTypeArraySize
468                                    )
469                                    :
470                                    new JavaCodeModel.ObjectType(null)
471                    );
472                    returnType = (
473                            returnTypeIsArray
474                                    ?
475                            new JavaCodeModel.ArrayType( new JavaCodeModel.ObjectType(qualifiedType), returnTypeArraySize)
476                                    :
477                            new JavaCodeModel.ObjectType(qualifiedType)
478                    );
479                } else { // Void type.
480                    returnType = JavaCodeModel.VOID;
481                }
482    
483                // Extract formal parameters.
484                List formalParameters = methodDeclaration.findChildrenOfType(ASTFormalParameter.class);
485                ASTFormalParameter formalParameter;
486                List<String> parameterNames = new ArrayList();
487                List<JavaCodeModel.JType> parameterTypes = new ArrayList();
488                for (int p = 0; p < formalParameters.size(); p++) {
489                    formalParameter = (ASTFormalParameter) formalParameters.get(p);
490                    // Parameter identifier.
491                    parameterNames.add(((ASTVariableDeclaratorId) formalParameter.findChildrenOfType(ASTVariableDeclaratorId.class).get(0)).getName());
492                    // Parameter type.
493                    parameterTypes.add(extractFormalparameterType(importsContext, formalParameter));
494                }
495    
496                // Extract exceptions.
497                JavaCodeModel.ExceptionType[] exceptions = extractExceptions(importsContext, methodDeclaration);
498    
499                // Storing method.
500                methodsBuffer.add(
501                        new JavaSourceFileParser.Method(
502                                extractModifiers(methodDeclaration),
503                                methodVisibility,
504                                qualifyMethod(packagePath, methodName),
505                                parameterNames.toArray(new String[parameterNames.size()]),
506                                parameterTypes.toArray(new JavaCodeModel.JType[parameterTypes.size()]),
507                                returnType,
508                                exceptions
509    
510                        )
511                );
512            }
513    
514            // Group signatures by name.
515            Map<String, List<Method>> methodsMap = new HashMap<String, List<Method>>();
516            Iterator<Method> iter = methodsBuffer.iterator();
517            JavaSourceFileParser.Method curr;
518            while (iter.hasNext()) {
519                curr = iter.next();
520                List signatures = methodsMap.get(curr.methodPath);
521                if (signatures == null) {
522                    signatures = new ArrayList();
523                    methodsMap.put(curr.methodPath, signatures);
524                }
525                signatures.add(curr);
526            }
527    
528            // Store methods.
529            Iterator<Map.Entry<String, List<JavaSourceFileParser.Method>>> entries = methodsMap.entrySet().iterator();
530            Map.Entry<String, List<JavaSourceFileParser.Method>> entry;
531            while (entries.hasNext()) {
532                entry = entries.next();
533                for (int i = 0; i < entry.getValue().size(); i++) {
534                    ch.method(
535                            entry.getValue().get(i).modifiers,
536                            entry.getValue().get(i).visibility,
537                            entry.getKey(),
538                            i,
539                            entry.getValue().get(i).parameterNames,
540                            entry.getValue().get(i).parameterTypes,
541                            entry.getValue().get(i).returnType,
542                            entry.getValue().get(i).exceptions
543                    );
544                }
545            }
546        }

548        private void extractEnumerations(String packagePath, CodeHandler ch, SimpleNode simpleNode) {
549            List enumDeclarations = simpleNode.findChildrenOfType(ASTEnumDeclaration.class);
550            ASTEnumDeclaration enumDeclaration = null;
551            for(int ed = 0; ed < enumDeclarations.size(); ed++) {
552                enumDeclaration = (ASTEnumDeclaration) enumDeclarations.get(ed);
553                 JavaCodeModel.JVisibility visibility = retrieveVisibility(enumDeclaration);
554                ASTIdentifier indentifier = (ASTIdentifier) enumDeclaration.findChildrenOfType(ASTIdentifier.class).get(0);
555                String identifier = indentifier.getName();
556                List enumElements = enumDeclaration.findChildrenOfType(ASTEnumElement.class);
557                List elements = new ArrayList();
558                for(int ee = 0; ee < enumElements.size(); ee++ ) {
559                    elements.add( ((ASTEnumElement) enumElements.get(ee)).getName() );
560                }
561                ch.startEnumeration(
562                        extractModifiers(enumDeclaration),
563                        visibility,
564                        qualifyEnumeration(packagePath, identifier),
565                        (String[]) elements.toArray(new String[elements.size()])
566                );
567                //TODO: HIGH - extract fields + methods.
568                ch.endEnumeration();
569                elements.clear();
570            }
571        }

573        private JavaCodeModel.JType extractFormalparameterType(ImportsContext importsContext, ASTFormalParameter formalParameter) {
574    
575            // Retrieve array size.
576            int arraySize;
577            List referenceTypes = formalParameter.findChildrenOfType(ASTReferenceType.class);
578            if( referenceTypes.size() > 0 ) {
579                ASTReferenceType referenceType = (ASTReferenceType) referenceTypes.get(0);
580                arraySize = referenceType.getArrayCount();
581            } else {
582                arraySize = 0;
583            }
584    
585            // Retrieve type.
586            List primitiveType = formalParameter.findChildrenOfType(ASTPrimitiveType.class);
587            if (primitiveType.size() > 0) { // Result type is Primitive.
588                return JavaCodeModel.javaTypeToJType(((ASTPrimitiveType) primitiveType.get(0)).getName());
589            } else { // Result type is Object.
590                String typeName =  retrieveObjectName( formalParameter.findChildrenOfType(ASTIdentifier.class) );
591                if(arraySize > 0) {
592                    return new JavaCodeModel.ArrayType(
593                            new JavaCodeModel.ObjectType(
594                                qualifyType(
595                                        importsContext,
596                                        typeName,
597                                        new JavaCodeModel.ArrayType(new JavaCodeModel.ObjectType(null), arraySize)
598                                )
599                            ),
600                             arraySize
601                    );
602                } else {
603                    return new JavaCodeModel.ObjectType(
604                            qualifyType(
605                                importsContext,
606                                typeName,
607                                new JavaCodeModel.ObjectType(null)
608                            )
609                    );
610                }
611            }
612        }

614        private JavaCodeModel.ExceptionType[] extractExceptions(ImportsContext importsContext,AccessNode accessNode) {
615            List nameLists = accessNode.findChildrenOfType(ASTNameList.class);
616            JavaCodeModel.ExceptionType[] exceptions = null;
617            if(nameLists.size() > 0) {
618                ASTNameList nameList = (ASTNameList) nameLists.get(0);
619                List namesList = nameList.findChildrenOfType(ASTName.class);
620                String exceptionName;
621                exceptions = new JavaCodeModel.ExceptionType[namesList.size()];
622                for( int i = 0; i < namesList.size(); i++ ) {
623                    ASTName name = (ASTName) namesList.get(i);
624                    exceptionName = retrieveObjectName(name.findChildrenOfType(ASTIdentifier.class));
625                    exceptions[i] = new JavaCodeModel.ExceptionType(
626                        qualifyType(importsContext, exceptionName, new JavaCodeModel.ExceptionType(null))
627                    );
628                }
629            }
630            return exceptions;
631        }

649        public void parse(File file) throws ParserException, IOException {
650            FileInputStream fis = null;
651            try {
652                fis = new FileInputStream(file);
653            } catch (FileNotFoundException fnfe) {
654                throw new ParserException(fnfe, file.getAbsolutePath());
655            }
656            JavaParser jp = new JavaParser(fis);
657            ASTCompilationUnit ast = null;
658            try {
659                ast = jp.CompilationUnit();
660                processCompilationUnit(file.getAbsolutePath(), ast);
661            } catch (ParseException pe) {
662                throw new ParserException(pe, file.getAbsolutePath());
663            } finally {
664                fis.close();
665            }
666        }

668        /**
669           * Returns the relative depth of a descendant in respect with the parent.
670           * @param parent
671           * @param descendant
672           * @return
673           */
674          private int childRelativeDepth(Node parent, Node descendant) {
675              Node childParent = descendant;
676              int depth = 0;
677              while(childParent != parent) {
678                  childParent = childParent.jjtGetParent();
679                  depth++;
680                  if(childParent == null) {
681                      throw new IllegalStateException();
682                  }
683              }
684              return depth;
685          }

688           * Returns a list of descendant of simpleNode of type maching the given clazz
689           * where all results are at the same depth level of the first child found.
690           *
691           * @param simpleNode
692           * @param clazz
693           * @return
694           */
695          private List findChildrenAtSameLevel(SimpleNode simpleNode, Class clazz) {
696              List<SimpleNode> children = simpleNode.findChildrenOfType(clazz);
697              if(children.size() == 0) {
698                  return children;
699              }
700              final int depthLevel = childRelativeDepth(simpleNode, children.get(0));
701              List sameDepth = new ArrayList();
702              for(SimpleNode child : children) {
703                  if( childRelativeDepth(simpleNode, child ) == depthLevel ) {
704                      sameDepth.add(child);
705                  }
706              }
707              return sameDepth;
708          }

716          private String retrieveObjectName(List list) {
717              String type = "";
718              for(int i = 0; i < list.size(); i++) {
719                  type += ( (ASTIdentifier) list.get(i) ).getName() + ( i < list.size() - 1 ? "." : "");

724          /**
725           * Returns the JVisibility of an access node.
726           *
727           * @param an
728           * @return
729           */
730          private JavaCodeModel.JVisibility retrieveVisibility(AccessNode an) {
731              if (an.isPublic()) {
732                  return JavaCodeModel.JVisibility.PUBLIC;
733              } else if (an.isProtected()) {
734                  return JavaCodeModel.JVisibility.PROTECTED;
735              } else if (an.isPackage()) {
736                  return JavaCodeModel.JVisibility.DEFAULT;
737              } else {
738                  return JavaCodeModel.JVisibility.PRIVATE;
739              }
740          }

742          /**
743           * Retruns the qualified typeName for a typeName string.
744           *
745           * @param typeName
746           * @return
747           */
748          private String qualifyType(ImportsContext importsContext, String typeName, JavaCodeModel.JType type) {
749              String qualifiedObject = importsContext.qualifyType(getObjectsTable(), typeName);
750              if (qualifiedObject == null) {
751                  String tempId = codeHandler.generateTempUniqueIdentifier();
752                  if(type instanceof JavaCodeModel.ObjectType) {
753                      ((JavaCodeModel.ObjectType) type).setInternalIdentifier(tempId);
754                  } else if( type instanceof JavaCodeModel.InterfaceType) {
755                      ((JavaCodeModel.InterfaceType) type).setInternalIdentifier(tempId);
756                  } else if( type instanceof JavaCodeModel.ArrayType) {
757                      ((JavaCodeModel.ArrayType) type).setInternalIdentifier(tempId);
758                  } else {
759                      throw new IllegalStateException();
760                  }
761                  getObjectsTable().addUnresolvedType(typeName, type, importsContext);
762                  return tempId;
763              }
764              return qualifiedObject;
765          }
